const digamma = require('math-digamma');

async function tfDigamma(tensor) {
    const tensorData = await tensor.data();
    const digammaData = [];
    for (const scalarData of tensorData) {
        digammaData.push(digamma(scalarData));
    }
    return tf.tensor(digammaData, tensor.shape, tensor.dtype);
